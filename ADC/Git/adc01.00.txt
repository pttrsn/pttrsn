#Atividade de Casa 01.00 - Git

git init         # Cria um novo repositório local

git clone [url]    # Clona um repositório remoto

git add [arquivo]  # Adiciona alterações ao stage

git commit -m ‘message’  # Salva alterações no histórico

git push          # Envia alterações para o repositório remoto

git pull             # Atualiza o repositório local com alterações remotas

git rebase # Aplica alterações de uma branch em outra

git rebase # Mescla duas branches

git log #Exibe o histórico de commits
